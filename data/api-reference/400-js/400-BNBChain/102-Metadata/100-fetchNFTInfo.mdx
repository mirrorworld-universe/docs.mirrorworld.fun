# `fetchNFTInfo`

Fetch BNB Chain NFT Info

```js
import { type QueryEVMNFTInfoPayload } from "@usemirrorworld/web3.js"

const payload: QueryEVMNFTInfoPayload = {
  contract: "0x011135F655e0916345C5188aA14fD4E2FEf1845E",
  token_id: 1,
}

await mirrorworld.BNBChain.Metadata.fetchNFTInfo(payload)
```
