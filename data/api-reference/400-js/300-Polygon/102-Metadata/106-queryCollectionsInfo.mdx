# `queryCollectionsInfo`

Search NFT Collections Info

```ts
import { type SearchCollectionsInfoV2 } from "@usemirrorworld/web3.js"

const payload: SearchCollectionsInfoV2 = {
  collections: [
    "0x71bd707A8007Df1f212B2e004c0a331F5b2ebD80",
    "0x38d0769Ee60ABA07016753022Bda3CBf8b9F9Fff",
  ],
}

await mirrorworld.Polygon.Metadata.queryCollectionsInfo(payload)
```
