# `fetchNFTsInfo`

Fetch Polygon NFTs Info Accepts Filtering Metadata

```ts
import {
  type QueryEVMNFTsPayloadV2,
  QueryNFTsFilterType,
} from "@mirrorworld/web3.js"

const payload: QueryEVMNFTsPayloadV2 = {
  contract: "0x011135F655e0916345C5188aA14fD4E2FEf1845E",
  page: 1,
  page_size: 10,
  order: {
    order_by: "price",
    desc: true,
  },
  sale: true,
  filter: [
    {
      filter_name: "color",
      filter_type: QueryNFTsFilterType.enum,
      filter_value: ["blue", "red"],
    },
    {
      filter_name: "attack",
      filter_type: QueryNFTsFilterType.range,
      filter_value: [1, 10],
    },
  ],
}

await mirrorworld.Polygon.Metadata.fetchNFTsInfo(payload)
```
