# `fetchNFTInfo`

Fetch Polygon NFT Info

```js
import { type QueryEVMNFTInfoPayload } from "@mirrorworld/web3.js"

const payload: QueryEVMNFTInfoPayload = {
  contract: "0x011135F655e0916345C5188aA14fD4E2FEf1845E",
  token_id: 1,
}

await mirrorworld.Polygon.Metadata.fetchNFTInfo(payload)
```
