# `queryCollectionsInfo`

Search NFT Collections Info

```ts
import { type SearchCollectionsInfoV2 } from "@mirrorworld/web3.js"

const payload: SearchCollectionsInfoV2 = {
  collections: ["3zPzwhsbwwD4z9NTNKzi3C7dPXJnxHzFiq459W2zVqQo"],
}

await mirrorworld.Solana.Metadata.queryCollectionsInfo(payload)
```
