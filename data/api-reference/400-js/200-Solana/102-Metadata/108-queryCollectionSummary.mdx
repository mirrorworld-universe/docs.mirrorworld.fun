# `queryCollectionSummary`

Search NFT Collection Filter Parameters

```ts
import { type QueryCollectionsSummaryV2 } from "@usemirrorworld/web3.js"

const payload: QueryCollectionsSummaryV2 = {
  collections: ["EDJPXGueNek8S3QXiTYfqhuu2egF6SPzeSJQbjFV1Ny1"],
}

await mirrorworld.Solana.Metadata.queryCollectionSummary(payload)
```
