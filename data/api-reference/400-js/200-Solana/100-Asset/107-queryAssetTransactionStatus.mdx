# `queryAssetTransactionStatus`

Queries The Transaction Status Of A Solana Asset By The Transaction Signature

```ts
import { type QueryAssetTransactionStatusPayload } from "@usemirrorworld/web3.js"

const payload: QueryAssetTransactionStatusPayload = {
  signatures: [
    "3yzxWp7FiYHS7Yfmvqc1AjBdWWNzKxZebp7psxiAC4LRda8iGEaFXYXu97E2886AtcCc8jHxc4ZUwztpK8SNVS2w",
    "4wgQ3xsAvELAwCgUFirDwqVYKj74nfUxbRY2zNJQnH3kKTyjYsJonbCo9kEJU4snjaM7DpqeosFqmudrhhF8d35C",
  ],
}

await mirrorworld.Solana.Asset.queryAssetTransactionStatus(payload)
```
