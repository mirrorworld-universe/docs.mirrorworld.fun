# `transferNFT`

Transfer An NFT On Solana

```ts
import { type ITransferSolanaNFTPayloadV2 } from "@mirrorworld/web3.js"

const payload: ITransferSolanaNFTPayloadV2 = {
  // Mint address
  mint_address: "DJhPp8Lgmvxdfy4SsRQWZD4gk3xtx5Uhs4dzBTDKTx6h",
  // Recipient address
  to_wallet_address: "9gCRJbdzntoTmGnpzAcp49aPi6HL557mKzWeVMcKuJYH",
}

await mirrorworld.Solana.Asset.transferNFT(payload)
```
